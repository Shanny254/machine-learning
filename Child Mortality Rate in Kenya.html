# Define risk categories based on mortality value
def risk_level(value):
    if value < 20:
        return 'Low'
    elif 20 <= value < 35:
        return 'Medium'
    else:
        return 'High'

df['risk_category'] = df['Value'].apply(risk_level)

df['mortality_rate_change'] = df.groupby('indicator')['Value'].diff()

df['significant_change'] = df['mortality_rate_change'].apply(lambda x: 'Significant Change' if abs(x) > 10 else 'Stable')

# Filter data for significant changes
significant_changes = df[df['significant_change'] == 'Significant Change']

# Plot the overall trend
plt.figure(figsize=(12, 6))
sns.lineplot(data=df, x='Date', y='Value', hue='indicator', marker='o', alpha=0.6)

# Highlight significant changes
plt.scatter(significant_changes['Date'], significant_changes['Value'], 
            color='red', s=100, label='Significant Change', edgecolor='black')

# Adding labels and title
